Judicial Decision-making based on Interpretations of Science (JuDIS)

Project: Judicial Decision-making based on Interpretations of Science

Abstract: The present study surveyed judges to examine how they consider and apply scientific information during sentencing determinations. Our results provided unique insights into the ways in which judges assess cases and consider sentencing outcomes when presented with scientific information to explicate defendants' behavior.

Data: This demo uses data from interviews with 34 U.S. criminal court judges that I collected in 2023.

This project was published as:

  1. Influence of scientific evidence on judges views and sentencing decision-making: Natural Language Processing and qualitative analysis.
  2. Bio-behavioral scientific evidence alters judges' sentencing decision-making: A quantitative analysis.

Please note that this notebook contains only a small example from the analyses performed for this project. The purpose of this notebook is to demonstrate the thought process and types of analyses performed and is not exhaustive or complete. The full analyses performed for each project are reported in the published article.

# Define stop words
stop_words = set(stopwords.words('english'))

# Define contraction mapping dictionary
contractions_mapping = {
    "i'm": "i am",
    "i've": "i have",
    "i'll": "i will",
    "we're": "we are",
    "we've": "we have",
    "we'll": "we will",
    "you're": "you are",

    "you'll": "you will",
    "he's": "he is",
    "he'll": "he will",
    "she's": "she is",
    "she'll": "she will",
    "it's": "it is",
    "it'll": "it will",
    "they're": "they are",
    "they've": "they have",
    "they'll": "they will",
}

# Function to expand contractions
def expand_contractions(text, contractions_mapping):
    pattern = re.compile(r'\b(' + '|'.join(contractions_mapping.keys()) + r')\b')
    def expand_match(contraction):
        return contractions_mapping[contraction.group(0)]
    expanded_text = pattern.sub(expand_match, text)
    return expanded_text

# Function to clean and preprocess text
def clean_text(text):
    # Remove non-alphabetic characters
    text = re.sub('[^A-Za-z]', ' ', text)
    # Convert to lowercase
    text = text.lower()
    # Expand contractions
    text = expand_contractions(text, contractions_mapping)
    # Tokenize the text
    tokens = word_tokenize(text)
    # Remove stopwords
    cleaned_text = ' '.join([word for word in tokens if word not in stop_words])
    return cleaned_text

#  The polarity score ranges from -1 to 1, where:
#  Negative values indicate negative sentiment.
#  Positive values indicate positive sentiment.
#  Zero indicates a neutral sentiment.

## Sentiment Analysis (TextBlob)
def analyze_sentiment(text):
    blob = TextBlob(text)
    sentiment_score = blob.sentiment.polarity
    if sentiment_score > 0:
        return 'Positive'
    elif sentiment_score < 0:
        return 'Negative'
    else:
        return 'Neutral'

# Function to count word frequencies
def count_word_frequencies(df, group_col):
    word_counts = {}
    for group, group_df in df.groupby(group_col):
        transcripts = group_df['Cleaned_Transcript']
        words = ' '.join(transcripts).split()
        word_counts[group] = {word: words.count(word) for word in set(words)}
    return word_counts

## Clean and preprocess text data

# Get English stop words list
stop_words = set(stopwords.words('english'))

# Custom removal of words
custom_stop_words = ['think', 'know', 're', 'would', 'going', 'they', 'things', 'got',
                     'get', 'want', 'way', 'try', 'lot', 'make', 'see', 'go', 'may',
                     'say', 'something', 'people', 'person', 'really', 'someone', 'come',
                     'came', 'small', 'kind', 'could', 'even', 'like', 'much', 'less',
                     'hear', 'worse', 'sense', 'time', 'well', 'consider', 'still', 'give',
                     'na', 'gon','sides','situation', 'seeing', 'let', 'weeks', 'somebody',
                     'put', 'one', 'two', 'full','front', 'play', 'never', 'whether',
                     'leave', 'cars', 'generally', 'place', 'compared', 'found',
                     'background', 'us', 'gonna', 'thing', 'issue', 'issues', 'factor']
stop_words = stop_words.union(custom_stop_words)
stop_words = list(stop_words)

# Define contraction mapping dictionary
contractions_mapping = {
    "i'm": "i am", "i've": "i have", "i'll": "i will", "we're": "we are",
    "we've": "we have", "we'll": "we will", "you're": "you are", "you've": "you have",
    "you'll": "you will", "he's": "he is", "he'll": "he will", "she's": "she is",
    "she'll": "she will", "it's": "it is", "it'll": "it will",
    "they're": "they are", "they've": "they have", "they'll": "they will" }

# Function to expand contractions
def expand_contractions(text, contractions_mapping):
    pattern = re.compile(r'\b(' + '|'.join(contractions_mapping.keys()) + r')\b')
    def expand_match(contraction):
        return contractions_mapping[contraction.group(0)]
    expanded_text = pattern.sub(expand_match, text)
    return expanded_text

# Function to clean and preprocess text
def clean_text(text):
    tokens = word_tokenize(text)# Tokenize the text
    text = re.sub('[^A-Za-z]', ' ', text) # Remove non-alphabetic characters
    text = text.lower() # Convert to lowercase
    text = expand_contractions(text, contractions_mapping) # Expand contractions
    text = re.sub(r'[^\w\s]', ' ', text.lower()) # Remove punctuation and convert to lowercase
    cleaned_text = ' '.join([word for word in tokens if word not in stop_words]) # Remove stopwords
    return cleaned_text

# Apply cleaning
dfQUAL['TranscriptFull'] = dfQUAL['TranscriptFull'].apply(clean_text)
